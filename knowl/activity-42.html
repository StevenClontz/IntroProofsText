\(\exists y\ \forall x, x < y\text{.}\)

\(\exists x\ \forall y, x < y\text{.}\)
(d)
\(\forall y\ \exists x, x < y\text{.}\)
(e)
\(\exists x\ \exists y, x < y\text{.}\)
(f)
\(\forall x\ \forall y, x < y\text{.}\)
